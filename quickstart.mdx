## Get Started

<Steps>
  <Step title="Setup your app" stepNumber={1}>
    create a new app

    ```shellscript
    npm create vite@latest my-pure-react-app --template react-ts
    ```

    or, open your existing app

    ```shellscript
    cd your-app
    ```
  </Step>
  <Step title="Install the SDK" stepNumber={2}>
    <Tabs>
      <Tab title="npm">
        ```shellscript
        npm install @auth0/auth0-react
        ```
      </Tab>
      <Tab title="yarn">
        ```shellscript
        yarn install @auth0/auth0-react
        ```
      </Tab>
      <Tab title="pnpm">
        ```shellscript
        pnpm install @auth0/auth0-react
        ```
      </Tab>
    </Tabs>
